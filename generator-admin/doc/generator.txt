


模板变量
%s: 文件名前缀
${g.codePackage}: 包名
${g.name}: 文件备注/描述
${g.author}: 作者
${g.date}: 日期时间
${g.tableName}: 表名
${field.fieldName}: 列名
${g.entityName}: 实体名
${field.comment}: 字段备注
${field.chinaName}: 中文名，一般不用
${field.type}: 字段类型
${field.name}: 字段名

# 控制层路径/变量名，第一个字母小写
${strutil.toLowerCase(strutil.subStringTo(g.entityName,0,1))}${strutil.subString(g.entityName,1)}

# 转小写
${strutil.toLowerCase(g.entityName)}


# 实体示例:
<%for(field in t.fields){%>
     /**
     * ${field.chinaName}
     * ${field.comment}
     */

	private ${field.type} ${field.name};

<%}%>
